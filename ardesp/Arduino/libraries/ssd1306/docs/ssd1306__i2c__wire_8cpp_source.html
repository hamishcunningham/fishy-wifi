64 void ssd1306_i2cSendByte_Wire(uint8_t data)
65 {
66  // Do not write too many bytes for standard Wire.h. It may become broken
67 #if defined(ESP32) || defined(ESP31B)
68  if (s_bytesWritten >= (I2C_BUFFER_LENGTH >> 4))
69 #elif defined(ARDUINO_ARCH_SAMD)
70  if (s_bytesWritten >= 64)
71 #elif defined(BUFFER_LENGTH)
72  if (s_bytesWritten >= (BUFFER_LENGTH >> 1))
73 #else
74  if (s_bytesWritten >= (USI_BUF_SIZE -2))
75 #endif
76  {
77  ssd1306_i2cStop_Wire();
78  ssd1306_i2cStart_Wire();
79  /* Commands never require many bytes. Thus assume that user tries to send data */
80  Wire.write(0x40);
81  s_bytesWritten++;
82  }
83  Wire.write(data);
84  s_bytesWritten++;
85 }
