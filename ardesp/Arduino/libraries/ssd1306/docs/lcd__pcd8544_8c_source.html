46 static void pcd8544_setBlock(uint8_t x, uint8_t y, uint8_t w)
47 {
48  s_width = w;
49  s_column = x;
50  s_page = y;
51  ssd1306_commandStart();
52  if (w == 1) ssd1306_sendByte( 0x22 ); else ssd1306_sendByte( 0x20 );
53  ssd1306_sendByte(0x80 | x);
54  ssd1306_sendByte(0x40 | y);
55  ssd1306_spiDataMode(1);
56 }

58 static void pcd8544_nextPage(void)
59 {
60  if ( s_width != 1)
61  {
62  ssd1306_endTransmission();
63  pcd8544_setBlock(s_column, s_page+1, s_width);
64  }
65 }

67 void pcd8544_84x48_init()
68 {
69  g_lcd_type = LCD_TYPE_PCD8544;
70  s_displayWidth = 84;
71  s_displayHeight = 48;
72  ssd1306_commandStart();
73  ssd1306_setRamBlock = pcd8544_setBlock;
74  ssd1306_nextRamPage = pcd8544_nextPage;
75  ssd1306_sendPixels = ssd1306_sendByte;
76  for( uint8_t i=0; i<sizeof(s_lcd84x48_initData); i++)
77  {
78  ssd1306_sendByte(pgm_read_byte(&s_lcd84x48_initData[i]));
79  }
80  ssd1306_endTransmission();
81 }

83 void pcd8544_84x48_spi_init(int8_t rstPin, int8_t cesPin, int8_t dcPin)
84 {
85  if (rstPin >=0)
86  {
87  pinMode(rstPin, OUTPUT);
88  digitalWrite(rstPin, HIGH);
89  /* Wait at least 30ms after VCC is up for LCD */
90  delay(30);
91  /* Perform reset operation of LCD display */
92  digitalWrite(rstPin, LOW);
93  delay(1);
94  digitalWrite(rstPin, HIGH);
95  }
96  ssd1306_spiInit(cesPin, dcPin);
97  pcd8544_84x48_init();
98 }
