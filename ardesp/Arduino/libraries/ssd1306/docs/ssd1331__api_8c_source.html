45 void ssd1331_drawLine(uint8_t x1, uint8_t y1, uint8_t x2, uint8_t y2, uint16_t color)
46 {
47  ssd1306_commandStart();
48  ssd1306_sendByte(SSD1331_DRAWLINE);
49  ssd1306_sendByte(x1);
50  ssd1306_sendByte(y1);
51  ssd1306_sendByte(x2);
52  ssd1306_sendByte(y2);
53  ssd1306_sendByte( (color & 0x03) << 4 );
54  ssd1306_sendByte( (color & 0x1C) << 2 );
55  ssd1306_sendByte( (color & 0xE0) >> 2 );
56  ssd1306_endTransmission();
57 }

59 void ssd1331_fastDrawBuffer8(uint8_t x, uint8_t y, uint8_t w, uint8_t h, const uint8_t *data)
60 {
61  uint16_t count = w * h;
62  ssd1306_setRamBlock(x, y, w);
63  while (count--)
64  {
65  ssd1306_sendPixel8( *data );
66  data++;
67  }
