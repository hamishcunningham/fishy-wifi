196  inline SSD1306_RECT getUpdateRect() const
197  {
198  uint8_t left = min(x,lx);
199  uint8_t top = min(y,ly);
200  uint8_t right = max((uint8_t)(x + w - 1), (uint8_t)(lx + w - 1));
201  if (((uint8_t)(lx + w - 1) < w) && (right > 2*w))
202  {
203  right = (uint8_t)(lx + w - 1);
204  }
205  uint8_t bottom = max((uint8_t)(y + 7), (uint8_t)(ly + 7));
206  if (((uint8_t)(ly + 7) < 8) && (bottom > 16))
207  {
208  bottom = (uint8_t)(ly + 7);
209  }
210  if ( left > right ) left = 0;
211  if ( top > bottom ) top = 0;
