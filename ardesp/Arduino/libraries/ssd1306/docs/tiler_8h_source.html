104  static void refresh(const NanoPoint &point)
105  {
106  if ((point.y<0) || ((point.y>>B)>=NE_MAX_TILES_NUM)) return;
107  m_refreshFlags[(point.y>>B)] |= (1<<(point.x>>B));
108  }

114  static void refresh(lcdint_t x1, lcdint_t y1, lcdint_t x2, lcdint_t y2)
115  {
116  y1 = max(0,y1>>B);
117  y2 = min((y2>>B), NE_MAX_TILES_NUM - 1);
118  for(uint8_t y=y1; y<=y2; y++)
119  {
120  for(uint8_t x=max(0,(x1>>B)); x<=(x2>>B); x++)
121  {
122  m_refreshFlags[y] |= (1<<x);
123  }
124  }
125  }

187 template<class C, uint8_t W, uint8_t H, uint8_t B>
188 void NanoEngineTiler<C,W,H,B>::displayBuffer()
189 {
190  if (!m_onDraw) // If onDraw handler is not set, just output current canvas
191  {
192  canvas.blt();
193  return;
194  }
195  for (uint8_t y = 0; y < s_displayHeight; y = y + NE_TILE_HEIGHT)
196  {
197  uint16_t flag = m_refreshFlags[y >> NE_TILE_SIZE_BITS];
198  m_refreshFlags[y >> NE_TILE_SIZE_BITS] = 0;
199  for (uint8_t x = 0; x < s_displayWidth; x = x + NE_TILE_WIDTH)
200  {
201  if (flag & 0x01)
202  {
203  canvas.setOffset(x, y);
204  if (m_onDraw()) canvas.blt();
205  }
206  flag >>=1;
207  }
208  }
209 }

211 template<class C, uint8_t W, uint8_t H, uint8_t B>
212 void NanoEngineTiler<C,W,H,B>::displayPopup(const char *msg)
213 {
214  NanoRect rect = { 8, (s_displayHeight>>1) - 8, s_displayWidth - 8, (s_displayHeight>>1) + 8 };
215  // TODO: It would be nice to calculate message height
216  NanoPoint textPos = { (s_displayWidth - (lcdint_t)strlen(msg)*s_fixedFont.width) >> 1, (s_displayHeight>>1) - 4 };
217  refresh(rect);
218  for (uint8_t y = 0; y < s_displayHeight; y = y + NE_TILE_HEIGHT)
219  {
220  uint16_t flag = m_refreshFlags[y >> NE_TILE_SIZE_BITS];
221  m_refreshFlags[y >> NE_TILE_SIZE_BITS] = 0;
222  for (uint8_t x = 0; x < s_displayWidth; x = x + NE_TILE_WIDTH)
223  {
224  if (flag & 0x01)
225  {
226  canvas.setOffset(x, y);
227  if (m_onDraw) m_onDraw();
228  canvas.setColor(RGB_COLOR8(0,0,0));
229  canvas.fillRect(rect);
230  canvas.setColor(RGB_COLOR8(192,192,192));
231  canvas.drawRect(rect);
232  canvas.printFixed( textPos.x, textPos.y, msg);
233 
234  canvas.blt();
235  }
236  flag >>=1;
237  }
238  }
239 }
