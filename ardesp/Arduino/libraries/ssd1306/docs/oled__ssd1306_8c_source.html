73 static void ssd1306_setBlock(uint8_t x, uint8_t y, uint8_t w)
74 {
75  ssd1306_commandStart();
76  ssd1306_sendByte(SSD1306_COLUMNADDR);
77  ssd1306_sendByte(x);
78  ssd1306_sendByte(w ? (x + w - 1) : (s_displayWidth - 1));
79  ssd1306_sendByte(SSD1306_PAGEADDR);
80  ssd1306_sendByte(y);
81  ssd1306_sendByte((s_displayHeight >> 3) - 1);
82  if (ssd1306_dcQuickSwitch)
83  {
84  ssd1306_spiDataMode(1);
85  }
86  else
87  {
88  ssd1306_endTransmission();
89  ssd1306_dataStart();
90  }
91 }

150 void ssd1306_128x64_spi_init(int8_t rstPin, int8_t cesPin, int8_t dcPin)
151 {
152  if (rstPin >=0)
153  {
154  pinMode(rstPin, OUTPUT);
155  digitalWrite(rstPin, HIGH);
156  /* Wait at least 1ms after VCC is up for LCD */
157  delay(1);
158  /* Perform reset operation of LCD display */
159  digitalWrite(rstPin, LOW);
160  delay(10);
161  digitalWrite(rstPin, HIGH);
162  }
163  ssd1306_spiInit(cesPin, dcPin);
164  ssd1306_128x64_init();
165 }
